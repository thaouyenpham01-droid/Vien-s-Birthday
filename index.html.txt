<script>
const fl=document.getElementById('floaters');
const items=['⭐','✨','💫','🌟','💕','🎀','🌸','💖','🎈'];
for(let i=0;i<20;i++){
  const f=document.createElement('div');f.className='floater';
  f.textContent=items[Math.floor(Math.random()*items.length)];
  const dur=7+Math.random()*8;
  f.style.cssText='left:'+(Math.random()*100)+'%;font-size:'+(12+Math.random()*14)+'px;animation-duration:'+dur+'s;animation-delay:'+(Math.random()*dur)+'s;';
  fl.appendChild(f);
}
const canvas=document.getElementById('confetti');
const ctx=canvas.getContext('2d');
canvas.width=window.innerWidth;canvas.height=window.innerHeight;
const cols=['#ff69b4','#ffd700','#ff1493','#fff','#cc1111','#ffb6d9','#ffe066'];
const pieces=[];
for(let i=0;i<70;i++){
  pieces.push({x:Math.random()*canvas.width,y:Math.random()*canvas.height-canvas.height,w:Math.random()*9+3,h:Math.random()*6+2,color:cols[Math.floor(Math.random()*cols.length)],vx:(Math.random()-0.5)*2,vy:Math.random()*1.5+0.5,rot:Math.random()*360,vrot:(Math.random()-0.5)*4});
}
function draw(){
  ctx.clearRect(0,0,canvas.width,canvas.height);
  pieces.forEach(p=>{
    p.x+=p.vx;p.y+=p.vy;p.rot+=p.vrot;
    if(p.y>canvas.height){p.y=-10;p.x=Math.random()*canvas.width;}
    ctx.save();ctx.translate(p.x,p.y);ctx.rotate(p.rot*Math.PI/180);
    ctx.globalAlpha=0.75;ctx.fillStyle=p.color;
    ctx.fillRect(-p.w/2,-p.h/2,p.w,p.h);ctx.restore();
  });
  requestAnimationFrame(draw);
}
draw();
</script>
